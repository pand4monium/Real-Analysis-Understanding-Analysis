\section{Compact Sets} 

\bx{ % Exercise 3.3.1
    Show that if $K$ is compact and nonempty, then $\sup K$ and $\inf K$ both exist and are elements of $K$.
    \sol{
        Since $K$ is bounded; as it is compact, both $\sup K$ and $\inf K$ exists.

        Let $s =\sup K$, then by definition, for $\epsilon > 0$, we can find $x \in K$, with $s -\epsilon < x$. Picking $\epsilon_n = 1/n$ and $x_n$, we get $(x_n) \rightarrow s$, thus $s \in K$; since $K$ is compact.
        A similar arguement can be used to prove the $\inf K \in K$.
    }
}

\bx{ % 3.3.2
    Decide which of the following sets are compact. For those that are not compact, show how Definition 3.3.1 breaks down. In other words, give an example of a sequence contained in the given set that does not possess a subsequence converging to a limit in the set.
    \ea{
        \item $\mathbb{N}$.
        \item $\mathbb{Q} \cap [0,1]$.
        \item The Cantor set.
        \item $\{1 + 1/2^2 + 1/3^2 + \dots + 1/n^2 : n \in \mathbb{N}\}$.
        \item ${1, 1/2, 2/3, 3/4, 4/5, \dots}$.
    }
    \sol{\ea{
        \item Not compact, as $\mathbb{N}$ is unbounded.
        \item Not compact, as we can find the sequence of first $n$ decimals of $\sqrt2 - 1$, ie. $\{0, 0.4, 0.41, 0.414, \dots\} \rightarrow \sqrt2 - 1 \notin \mathbb{Q} \cap [0,1]$.
        \item Compact, as it the countable intersection of compact sets.
        \item Not compact, as $\frac{\pi^2}{6}$ is not contained in the set, despite being a limit point.
        \item Compact, since it is bounded and closed as every subsequence converges to $1$, which is contained in the set.
    }}
}

\bx{ % 3.3.3
    Prove the converse of Theorem 3.3.4 by showing that if a set $K \subseteq \mathbb{R}$ is closed and bounded, then it is compact.
    \sol{
        Let $K \subseteq \mathbb{R}$ be closed and bounded, and let $(x_n)$ be a sequence contained within $K$. By Bolzano-Weierstrass Theorem, since $(x_n)$ is bounded; as $K$ is bounded, we can find a convergent subsequence $(x_{n_k}) \rightarrow x$. Since $K$ is closed, $x \in K$. Thus every sequence in $K$ contains a subsequence converging to a limit in $K$, which is the definition of being compact.
    }
}

\bx{ % 3.3.4
    Assume $K$ is compact and $F$ is closed. Decide if the following sets are definitely compact, definitely closed, both or neither.\ea{
        \item $K \cap F$
        \item $\overline{F^c \cup K^c}$
        \item $K \backslash F = \{x \in K : x \notin F\}$
        \item $\overline{K \cap F^c}$
    }
    \sol{\ea{
        \item Definitely closed and compact. Since $K$ is bounded, $K \cap F \subseteq K$ is also bounded. $K \cap F$ is the intersection of two closed set, hence also closed.
        \item Definitely closed and not compact. The closure of any set is always closed. However, it is definitely not compact as $K$ is bounded, so $K^c$ is unbounded, implying that $\overline{F^c \cup K^c}$ is unbounded.
        \item $K \backslash F = K \cap F^c$ could be either. While it is definitely bounded since it is the subset of $F$ which is bounded, it could be either open and closed.
        \item Definitely closed and compact. The closure of any set is always closed. The set is a subset of $K$ which is bounded, thus also being bounded.
    }}
}

\section{Compact Sets} 

\bx{ % Exercise 3.3.1
    Show that if $K$ is compact and nonempty, then $\sup K$ and $\inf K$ both exist and are elements of $K$.
    \sol{
        Since $K$ is bounded; as it is compact, both $\sup K$ and $\inf K$ exists.

        Let $s =\sup K$, then by definition, for $\epsilon > 0$, we can find $x \in K$, with $s -\epsilon < x$. Picking $\epsilon_n = 1/n$ and $x_n$, we get $(x_n) \rightarrow s$, thus $s \in K$; since $K$ is compact.
        A similar arguement can be used to prove the $\inf K \in K$.
    }
}

\bx{ % 3.3.2
    Decide which of the following sets are compact. For those that are not compact, show how Definition 3.3.1 breaks down. In other words, give an example of a sequence contained in the given set that does not possess a subsequence converging to a limit in the set.
    \ea{
        \item $\mathbb{N}$.
        \item $\mathbb{Q} \cap [0,1]$.
        \item The Cantor set.
        \item $\{1 + 1/2^2 + 1/3^2 + \dots + 1/n^2 : n \in \mathbb{N}\}$.
        \item ${1, 1/2, 2/3, 3/4, 4/5, \dots}$.
    }
    \sol{\ea{
        \item Not compact, as $\mathbb{N}$ is unbounded.
        \item Not compact, as we can find the sequence of first $n$ decimals of $\sqrt2 - 1$, ie. $\{0, 0.4, 0.41, 0.414, \dots\} \rightarrow \sqrt2 - 1 \notin \mathbb{Q} \cap [0,1]$.
        \item Compact, as it the countable intersection of compact sets.
        \item Not compact, as $\frac{\pi^2}{6}$ is not contained in the set, despite being a limit point.
        \item Compact, since it is bounded and closed as every subsequence converges to $1$, which is contained in the set.
    }}
}

\bx{ % 3.3.3
    Prove the converse of Theorem 3.3.4 by showing that if a set $K \subseteq \mathbb{R}$ is closed and bounded, then it is compact.
    \sol{
        Let $K \subseteq \mathbb{R}$ be closed and bounded, and let $(x_n)$ be a sequence contained within $K$. By Bolzano-Weierstrass Theorem, since $(x_n)$ is bounded; as $K$ is bounded, we can find a convergent subsequence $(x_{n_k}) \rightarrow x$. Since $K$ is closed, $x \in K$. Thus every sequence in $K$ contains a subsequence converging to a limit in $K$, which is the definition of being compact.
    }
}

\bx{ % 3.3.4
    Assume $K$ is compact and $F$ is closed. Decide if the following sets are definitely compact, definitely closed, both or neither.\ea{
        \item $K \cap F$
        \item $\overline{F^c \cup K^c}$
        \item $K \backslash F = \{x \in K : x \notin F\}$
        \item $\overline{K \cap F^c}$
    }
    \sol{\ea{
        \item Definitely closed and compact. Since $K$ is bounded, $K \cap F \subseteq K$ is also bounded. $K \cap F$ is the intersection of two closed set, hence also closed.
        \item Definitely closed and not compact. The closure of any set is always closed. However, it is definitely not compact as $K$ is bounded, so $K^c$ is unbounded, implying that $\overline{F^c \cup K^c}$ is unbounded.
        \item $K \backslash F = K \cap F^c$ could be either. While it is definitely bounded since it is the subset of $F$ which is bounded, it could be either open and closed.
        \item Definitely closed and compact. The closure of any set is always closed. The set is a subset of $K$ which is bounded, thus also being bounded.
    }}
}

\bx{ % 3.3.5
    Decide wheteher the following propositions are true or false. If the claim is valid, supply a short proof, and if the claim is false, provide a counterexample. \ea{
        \item The arbituary intersection of compact sets is compact. 
        \item The arbitrary union of compact sets is compact.
        \item Let $A$ be arbitrary, and let $K$ bere compact. Then, the intersection $A \cap K$ is compact.
        \item If $F_1 \supseteq F_2 \supseteq F_3 \supseteq F_4 \supseteq \cdots$ is a nested sequence of nonempty closed sets, then the intersection $\bigcap_{n=1}^\infty F_n \ne \emptyset$.
    }
    \sol{\ea{
        \item True. The arbitrary intersection of closed sets is closed, and the arbitrary intersection of bounded sets is also bounded. Thus, the arbitrary intersection of compact sets is both bounded and closed, hence compact.
        \item False. Consider the sets $I_n = [0,n]$, which is all closed and bounded for all $n \in \mathbb{N}$. However, the arbitrary union, $\bigcup_{i=1}^\infty I_n = [0, \infty)$, which is not bounded and hence not compact.
        \item False. Consider $A = (\frac12, 1)$ and $K = \mathbb R$, then the intersection $A \cap K = A $  which is open, thus not compact.
        \item False. Considering $F_n = [n, \infty)$ a sequence of closed sets, however, $\bigcap_{n=1}^\infty F_n = \emptyset$.
    }}
}

\bx{ % 3.3.6
    This exercise is meant to illustrate the point made in the opening paragraph to Section 3.3. Verify that the following statements are true if every blank is filled with the word "finite". Which are true if every blank is filled with the word "compact"? Which are ture if every blank is filled with the word "closed"?  \ea{
        \item Every \_\_\_\_ set has a maximum.
        \item If $A$ and $B$ are \_\_\_\_, then $A + B = \{ a+b: a\in A, b\in B\}$ is also \_\_\_\_.
        \item If $\{ A_n : n \in\mathbb N \}$ is a collection of  \_\_\_\_ sets with the property that every finite subcollection has a nonempty intersection, then $\bigcap_{n=1}^\infty A_n $ is nonempty as well.
    }
    \sol{\ea{
        \item \er{
            \item Every finite set has a maximum. This is trivial as we can choose the largest term to be the maximum.
            \item Every compact set has a maximum. The set is bounded and closed. Hence the supremum is the maximum of the set.
            \item Not every closed set has a maximum. The set $\mathbb R$ is closed, but does not have a maximum as it is unbounded above.
        }
        \item \er{
            \item Finite is true, as we can just enumerate the terms of $A$, $B$ and $A+B$.
            \item For compact sets, the bounded property is preserved as $\sup A _+ \sup B = \sup (A+B)$. We will now show that $A+B$ is also closed. Suppose we have $c \in A+B$, then we can find a convergent sequence $(a_n + b_n) \rightarrow c$. Since $A$ is closed and bounded, we can find $(a_n)$ that is bounded and converges to some $a \in A$,, and taking the corresponding elements in $(b_n)$, we have a new bounded sequence that converges to some $b \in B$. Taking the corresponding elemetns in $(a_n + b_n)$ shows that $a+b = c$, thus is also closed.
            \item This is not true for only closed functions. Consider the sets $A = \{ n : n\in \mathbb N\}$ and $B = \{ -n + 1/n : n \in \mathbb N\}$, then the sequence $(1/n) \rightarrow  0 \notin A + B$.
        }
        \item \er{
            \item Finite is true, as the only way for every finite subcollection to have a nonempty intersection is if there is one element that is included in every set.
            \item Compact is true. We can have a series of nested compact sets $K_n =\bigcap_{n=1}^\infty A_k$, giving $K_n \subseteq K_{n-1}$, with each $K_n$ being nonempty. Then by Nested Compact Set Property, the intersection is nonempty.
            \item Closed is false, using the sets $A_n = [n, \infty)$
        }
    }}
}

\bx{ % 3.3.7
    As some more evidence that the surprising nature of the Cantor Set, follow these steps to show that the sum $C + C = \{ x+y: x, y\in C \}$ is equal to the closed interval $[0,2]$. (Keep in mind that $C$ has zero length and contains no intervals.) 


    Because $C \subseteq [0,1]$, $C+ C \subseteq [0,2]$ , so we only need to prove the reverse inclusion $[0,2] \subseteq \{ x+y: x, y \in C\}$. Thus, given $s \in [0,2]$, we must find two elements $x,y \in C$ satisfying $x+y=s$. \ea{
        \item Show that there exists $x_1, y_1 \in C_1$ for which $x_1 + y_1 = s$. Show in general that, for any arbitrary $n\in\mathbb{ N}$, we can always find $x_n , y_n \in C_n$ for which $x_n + y_n = s$.
        \item Keeping in mind that the sequences $(x_n)$ and $(y_n)$ do not necessarily converge, show how they can nevertheless be used to produce the desired $x$ and $y$ in $C$ satisfying $x+y = s$.
    }
    \sol{\ea{
        \item Using the definition of $A +B = \{ a +b: a \in A , b\in B\}$ and the fact that $[a,b] + [c, d] = [a + c, b+d]$, recall that $C_1 = [0, 1/3] \cup  [2/3, 1]$, then \begin{align*}
            C_1 + C_1 & = ([0,1/3] \cup [2/3, 1]) + ([0,1/3] \cup [2/3,1]) \\
            & = ([0,1/3]+ [0,1/3]) \cup ([0, 1/3] + [2/3, 1]) \cup ([2/3, 1] \cup [2/3, 1]) \\
            & = [0, 2/3] \cup [2/3, 4/3] \cup [4/3, 2] = [0,2] 
        \end{align*}

        With the base case now proven, we will prove the remaining using induction. We will first observe that $C_{n+1}$ is made up of 2 identical copies of $C_n$. Define $c \cdot A = \{ ca : a \in A\}$ for $c \in \mathbb R$ and $A + c = \{ a+c : a \in A\}$ for $c\in\mathbb R$. Then we get $C_{n+1} = \frac13 C_n \cup (\frac 13 C_n + \frac 23)$, which we simplify to $3 C_{n+1} = C_n \cup (C_n + 2)$. Under the induction hypothesis of $C_n +C_n = [0,2]$,
        \begin{align*}
            3C_{n+1} + 3C_{n+1} & = (C_n +C_n) \cup (C_n + (C_n + 2)) \cup ((C_n + 2) + (C_n + 2))  \\ 
            & = [0,2] \cup [2, 4] \cup [4, 6] = [0,6] \\ 
            & \implies C_{n+1} + C_{n+1} = [0,2] 
        \end{align*}

        \item Since $C$ is compact, there exists a subsequence $(x_{n_k} ) \rightarrow x$ with $x \in C$ by BWT. Now since $x_{n_k} + y_{n_k} = s $ for all $k$, we have $\lim y_{n_k} = \lim s-x_{n_k} = s-x$. Now since $y_{n_k} \in C$, $y \in C$. Thus we have found $x, y \in C$, such that $x + y = s$.
    }}
}

\bx{ % 3.3.8
    Let $K$ and $L$ be nonempty compact sets, and define \[
    d = \inf \{ |x-y|: x \in K \text{ and } y \in L\}.\]
    This turns out to be a reasonable definition for the \textit{distance} between $K$ and $L$.\ea{
        \item If $K$ and $L$ are disjoint, show $d > 0$ and that $d = | x_0 - y_0 |$ for some $x_0 \in K$ and $ y_0 \in L$.
        \item Show that it's possible to have $d = 0$  if we assume that the disjoint sets $K$ and $L$ are closed.
    }
    \sol{\ea{
        \item The set $|K-L| = \{ |x-y|: x\in K, y\in L\}$ is compact as $K-L$ is compact and compactness is preserved through continuous functions such as $| \cdot |$. Thus $d = \inf |K-L|$ and $d = |x_0 - y_0|$ for some $x_0 \in K$ and $y_0 \in L$.
        \item Consider the set $K = \{ n : n \in \mathbb N \}$ and $ L = \{ n + 1/n : n \in \mathbb N \}$, which has $ d = 0$. Both sets are closed as the limits diverge.
    }}
}
